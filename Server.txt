import express from "express";
import cors from "cors";

const app = express();
const PORT = process.env.PORT || 3001;
const API_KEY = process.env.FOOTBALL_DATA_API_KEY;
const BASE_URL = "https://api.football-data.org/v4";

app.use(cors());
app.use(express.json());

function buildUrl(path, query = {}) {
  const url = new URL(`${BASE_URL}${path}`);
  for (const [key, value] of Object.entries(query)) {
    if (value !== undefined && value !== null && value !== "") {
      url.searchParams.set(key, String(value));
    }
  }
  return url;
}

async function proxyFootballData(path, req, res) {
  try {
    if (!API_KEY) {
      return res.status(500).json({ error: "FOOTBALL_DATA_API_KEY manquante" });
    }

    const url = buildUrl(path, req.query);

    const response = await fetch(url, {
      headers: {
        "X-Auth-Token": API_KEY
      }
    });

    const contentType = response.headers.get("content-type") || "";
    const payload = contentType.includes("application/json")
      ? await response.json()
      : await response.text();

    res.status(response.status).send(payload);
  } catch (error) {
    res.status(500).json({
      error: "Erreur proxy Football-Data",
      details: error.message
    });
  }
}

app.get("/api/worldcup/matches", (req, res) => proxyFootballData("/competitions/WC/matches", req, res));
app.get("/api/worldcup/teams", (req, res) => proxyFootballData("/competitions/WC/teams", req, res));
app.get("/api/worldcup/standings", (req, res) => proxyFootballData("/competitions/WC/standings", req, res));

app.listen(PORT, () => {
  console.log(`Proxy Football-Data actif sur http://localhost:${PORT}`);
});
